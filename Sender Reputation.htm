      def parse_raw_email(raw_message):
          lines=raw_message.split('\n')
          email = {}
          message =''
          
          keys_to_extract = ['from','to']
          for line in lines:
              if ':' not in line:
                  message +=line.strip()
                  email['body'] =message
              else:
                  pairs = line.split(':')
                  key = pairs[0].lower()
                  val = pairs[1].strip()
                  if key in keys_to_extract:
                      email[key] = val
          return email

      def map_to_list(emails,key):
          results=[]
          for email in emails:
              if key not in email:
                  results.append('')
              else:
                  results.append(email[key])
          return results
